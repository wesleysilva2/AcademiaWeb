CREATE TABLE `usuarios` (
  `id` int(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `username` varchar(100) NOT NULL,
  `email` varchar(100) NOT NULL,
  `password` varchar(100) NOT NULL,
  `cursosCadastrados` varchar(100) NOT NULL,
  `academiaCadastrada` varchar(100) NOT NULL,
  `planoCadastrado` varchar(100) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1;

CREATE TABLE `plano` (
  `id` int(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `nomePlano` varchar(100) NOT NULL,
  `descricao` varchar(400) NOT NULL,
  `valor` float
) ENGINE=InnoDB DEFAULT CHARSET=latin1;

CREATE TABLE `cursos` (
  `id` int(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `nomeCurso` varchar(100) NOT NULL,
  `descricao` varchar(400) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1;


CREATE TABLE `academias` (
  `id` int(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `nomeAcademia` varchar(100) NOT NULL,
  `endereco` varchar(400) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1;

CREATE TABLE `cartao` (
  `id` int(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `cardName` varchar(100) NOT NULL,
  `cardNumber` varchar(100) NOT NULL,
  `cardValidade` varchar(100) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1;
